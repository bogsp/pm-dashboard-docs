# API Reference

Complete reference for all PM Dashboard components, utilities, and type definitions.

## Installation
```bash
npm install pm-dashboard
# or
yarn add pm-dashboard
# or
pnpm add pm-dashboard
```

## Components

### KanbanBoard

Visual task board with drag-and-drop columns.
```tsx
import { KanbanBoard } from 'pm-dashboard'

interface KanbanBoardProps {
  columns: Column[]
  onTaskMove?: (taskId: string, fromColumn: string, toColumn: string) => void
  onTaskClick?: (task: Task) => void
  onAddColumn?: () => void
  allowAddColumn?: boolean
  className?: string
}
```

**Example:**
```tsx
<KanbanBoard
  columns={columns}
  onTaskMove={(id, from, to) => updateTaskStatus(id, to)}
  onTaskClick={(task) => openTaskModal(task)}
/>
```

[Full documentation →](/docs/components/kanban-board)

  target: string

## Type Definitions

### Core Types
```tsx
// User
interface User {
  id: string
  name: string
  email: string
  avatar?: string
  role: 'owner' | 'member' | 'viewer'
  joinedAt: Date
}

// Project
interface Project {
  id: string
  name: string
  description?: string
  completion: number
  createdAt: Date
  updatedAt: Date
  owner: User
  members: User[]
  settings: ProjectSettings
}

// Task Status
type TaskStatus = 'todo' | 'in-progress' | 'done' | 'blocked'

// Task Priority
type TaskPriority = 'low' | 'medium' | 'high' | 'urgent'

// Activity Type
type ActivityType = 
  | 'task_created'
  | 'task_completed'
  | 'task_updated'
  | 'comment_added'
  | 'file_uploaded'
  | 'status_changed'
  | 'assignment_changed'
  | 'due_date_updated'

// File Type
type FileType = 'pdf' | 'image' | 'document' | 'spreadsheet' | 'archive' | 'other'

// Label
interface Label {
  id: string
  name: string
  color: string
}

// Comment
interface Comment {
  id: string
  taskId: string
  author: User
  content: string
  createdAt: Date
  updatedAt?: Date
  replies?: Comment[]
}

// Column
interface Column {
  id: string
  title: string
  tasks: Task[]
  color?: string
  limit?: number
}
```

## Utilities

### Date Formatting
```tsx
import { formatDate, formatRelativeTime } from 'pm-dashboard/utils'

// Format date
formatDate(new Date())  // "Jan 16, 2025"

// Relative time
formatRelativeTime(new Date(Date.now() - 3600000))  // "1 hour ago"
```

### Task Helpers
```tsx
import { 
  calculateCompletion,
  filterTasksByStatus,
  sortTasksByPriority,
  groupTasksByAssignee 
} from 'pm-dashboard/utils'

// Calculate completion percentage
const percentage = calculateCompletion(tasks)  // 67

// Filter by status
const inProgressTasks = filterTasksByStatus(tasks, 'in-progress')

// Sort by priority
const sortedTasks = sortTasksByPriority(tasks)

// Group by assignee
const tasksByAssignee = groupTasksByAssignee(tasks)
```

### File Helpers
```tsx
import { 
  formatFileSize,
  getFileType,
  validateFileType,
  generateThumbnail 
} from 'pm-dashboard/utils'

// Format file size
formatFileSize(2457600)  // "2.4 MB"

// Get file type from extension
getFileType('document.pdf')  // "pdf"

// Validate file type
validateFileType(file, ['pdf', 'png', 'jpg'])  // true/false

// Generate thumbnail for images
const thumbnail = await generateThumbnail(imageFile)
```

## Hooks

### useProject

Manage project state and operations.
```tsx
import { useProject } from 'pm-dashboard'

const {
  project,
  loading,
  error,
  updateProject,
  addMember,
  removeMember,
} = useProject(projectId)

// Update project
await updateProject({ name: 'New Name' })

// Add member
await addMember(userId, 'member')

// Remove member
await removeMember(userId)
```

### useTasks

Manage tasks within a project.
```tsx
import { useTasks } from 'pm-dashboard'

const {
  tasks,
  loading,
  createTask,
  updateTask,
  deleteTask,
  moveTask,
} = useTasks(projectId)

// Create task
await createTask({
  title: 'New Task',
  status: 'todo',
})

// Update task
await updateTask(taskId, { status: 'done' })

// Move task between columns
await moveTask(taskId, 'todo', 'in-progress')

// Delete task
await deleteTask(taskId)
```

### useActivities

Subscribe to project activities.
```tsx
import { useActivities } from 'pm-dashboard'

const {
  activities,
  loading,
  subscribe,
  unsubscribe,
} = useActivities(projectId)

// Subscribe to real-time updates
useEffect(() => {
  subscribe()
  return () => unsubscribe()
}, [])
```

### useFiles

Manage file uploads and downloads.
```tsx
import { useFiles } from 'pm-dashboard'

const {
  files,
  uploading,
  uploadFile,
  deleteFile,
  downloadFile,
} = useFiles(projectId)

// Upload file
const uploadedFile = await uploadFile(file)

// Download file
await downloadFile(fileId)

// Delete file
await deleteFile(fileId)
```

## Events

### Project Events
```tsx
import { ProjectEvents } from 'pm-dashboard'

// Listen to project events
ProjectEvents.on('project:updated', (project) => {
  console.log('Project updated:', project)
})

ProjectEvents.on('member:added', (member) => {
  console.log('New member:', member)
})

ProjectEvents.on('member:removed', (memberId) => {
  console.log('Member removed:', memberId)
})
```

### Task Events
```tsx
import { TaskEvents } from 'pm-dashboard'

// Listen to task events
TaskEvents.on('task:created', (task) => {
  console.log('Task created:', task)
})

TaskEvents.on('task:updated', (task) => {
  console.log('Task updated:', task)
})

TaskEvents.on('task:moved', ({ taskId, from, to }) => {
  console.log(`Task ${taskId} moved from ${from} to ${to}`)
})

TaskEvents.on('task:deleted', (taskId) => {
  console.log('Task deleted:', taskId)
})
```

        // Handle validation error
        break
      case ErrorCode.NOT_FOUND:
        // Handle not found

## Advanced Usage

### Custom Components

Extend base components with custom functionality:
```tsx
import { TaskCard as BaseTaskCard } from 'pm-dashboard'

const CustomTaskCard = ({ task, ...props }) => {
  return (
    <BaseTaskCard
      task={task}
      renderActions={(task) => (
        <CustomActions task={task} />
      )}
      {...props}
    />
  )
}
```

### Theming

Override default styles:
```tsx
import { ThemeProvider } from 'pm-dashboard'

const customTheme = {
  colors: {
    primary: '#8b5cf6',
    success: '#10b981',
    warning: '#f59e0b',
    danger: '#ef4444',
  },
  spacing: {
    xs: '0.25rem',
    sm: '0.5rem',
    md: '1rem',
    lg: '1.5rem',
    xl: '2rem',
  },
}

<ThemeProvider theme={customTheme}>
  <KanbanBoard columns={columns} />
</ThemeProvider>
```

## Migration Guide

### From v1.x to v2.x

**Breaking Changes:**

1. Renamed `TaskList` to `KanbanBoard`
2. Removed `ProjectCard` component
3. Updated `Activity` interface structure

**Migration:**
```tsx
// Before (v1.x)
import { TaskList } from 'pm-dashboard'
<TaskList tasks={tasks} />

// After (v2.x)
import { KanbanBoard } from 'pm-dashboard'
<KanbanBoard columns={columns} />
```
